server {
    ${ADD_SERVER_NAME}
    listen 80;
    root /usr/share/nginx/html;

    location / {
        try_files $uri$args $uri$args/ /index.html;
    }

    location ~* \.(eot|ttf|woff)$ {
        add_header Access-Control-Allow-Origin *;
    }

    ${ADD_PROXY_API}

    location ^~ /images/ {
        root /;
        autoindex on;
    }

    location ^~ /files/ {
        root /;
        autoindex on;
    }

}
